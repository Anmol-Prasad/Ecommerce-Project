#include <iostream>
using namespace std;

void pushZerosToEnd(int arr[], int n){
    int i=0;
    int j=n-1;
    while(i<j){
        while(arr[i]!=0){
            i++;
        }
        if(arr[j]!=0 && arr[i]==0){
            swap(arr[i], arr[j]);
            i++;
        }
        j--;
    }
    for(int m=0;m<n;m++){
        cout<<arr[m]<<" ";
    }
}

int main()
{
    int arr[] = {1, 9, 8, 4, 0, 0, 2, 7, 0, 6, 0, 9};
    int n = sizeof(arr) / sizeof(arr[0]);
    pushZerosToEnd(arr, n);
}
